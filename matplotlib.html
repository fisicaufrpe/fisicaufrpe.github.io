def ex2(x):
    """
    Funo
    """
    return 4*x**2 - 7*x +5

def diff_ex2(x):
    """
    Derivada da funo
    """
    return 8*x - 7

#####################################################
# valores de amostra 
#####################################################

x = np.linspace(-10, 10, 500) 
y = ex2(x)

#####################################################
# plotando o grfico
#####################################################

plt.plot(x, y, label="f(x)")
plt.plot(x, diff_ex2(x), label="df/dx")

#####################################################
# definindo o ponto mnimo
#####################################################

minimoX = 7/8 
minimoY = ex2(minimoX)

#####################################################
# exibindo o ponto 
#####################################################

plt.scatter(minimoX, minimoY, c="red") 

plt.annotate('Mnimo', # anotao
ha = 'center', va = 'bottom', # formatao da anotao
xytext = (minimoX-5, minimoY+200), #posio do texto
xy = (minimoX, minimoY), #posio inicial da seta
arrowprops = { 'facecolor' : 'black', 'shrink' : 0.05 }) #formatao da seta

plt.title("$4x^2-7x+5$") #ttulo do grfico
plt.legend() # insere legenda
plt.grid(True) #grade
plt.show() #exibe o grfico
