x,  y = [3,6,9,12,15],[2,4,6,8,10] #Valores de X e Y respectivamente para o grfico linha
plt.plot(x, y, c = "black") #Grfico estilo linha
plt.scatter(x, y, c = "red") #Grfico estilo pontos

plt.title("Ttulo") #Ttulo do grfico
plt.xlabel("Eixo X") #Ttulo para a varivel X
plt.ylabel("Eixo Y") #Ttulo para a varivel y
plt.grid(True) #coloca grades no grfico

plt.show() #Plota o grfico

x = np.linspace(-10, 10, 500) #Valores de amostra

plt.plot(x,np.sin(x), c = "green", label="sin") #Plota o grfico seno
plt.plot(x, np.cos(x), c = "blue", label = "cos") #Plota o grfico cosseno
plt.grid(True) #Insere a grade
plt.legend() #Insere a legenda 
plt.show() #Plota o grfico
