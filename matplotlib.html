x,  y = [3,6,9,12,15],[2,4,6,8,10] #Valores de X e Y respectivamente para o gráfico linha
plt.plot(x, y, c = "black") #Gráfico estilo linha
plt.scatter(x, y, c = "red") #Gráfico estilo pontos

plt.title("Título") #Título do gráfico
plt.xlabel("Eixo X") #Título para a variável X
plt.ylabel("Eixo Y") #Título para a variável y
plt.grid(True) #coloca grades no gráfico

plt.show() #Plota o gráfico

x = np.linspace(-10, 10, 500) #Valores de amostra

plt.plot(x,np.sin(x), c = "green", label="sin") #Plota o gráfico seno
plt.plot(x, np.cos(x), c = "blue", label = "cos") #Plota o gráfico cosseno
plt.grid(True) #Insere a grade
plt.legend() #Insere a legenda 
plt.show() #Plota o gráfico
